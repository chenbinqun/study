var div = document.getElementById('demo');
		var p = new Promise(function(resolve, reject){
			setTimeout(function(){
				resolve('promise创建');
			}, 2000);
		});
		
		p.then(function(val){
			div.innerHTML += ("<br />" + val);
			return "第一层then函数";
		})
		.then(function(val){
			div.innerHTML += ("<br />" + val);
			throw '第二层then出错啦';
		})
		.catch(function(err){
			div.innerHTML += ('<br />' + err)
		})
